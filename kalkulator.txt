/**
 * Klasa reprezentująca prosty kalkulator umożliwiający wykonanie operacji dodawania, odejmowania, mnożenia i dzielenia na dwóch liczbach.
 * Wykorzystuje bibliotekę Swing do wyświetlania okien dialogowych.
 */
public class Calculator {
    /**
     * Metoda główna programu, od której rozpoczyna się jego wykonanie.
     *
     * @param args Argumenty wiersza poleceń (nie są wykorzystywane w tym programie).
     */
    public static void main(String[] args) {
        // Pobranie pierwszej liczby od użytkownika
        String firstNumber = JOptionPane.showInputDialog("Podaj pierwszą liczbę:");
        double number1 = Double.parseDouble(firstNumber);

        // Pobranie drugiej liczby od użytkownika
        String secondNumber = JOptionPane.showInputDialog("Podaj drugą liczbę:");
        double number2 = Double.parseDouble(secondNumber);

        // Wyświetlenie opcji operacji matematycznych
        String[] options = {"Dodawanie", "Odejmowanie", "Mnożenie", "Dzielenie"};
        int choice = JOptionPane.showOptionDialog(
                null,
                "Wybierz operację:",
                "Kalkulator",
                JOptionPane.DEFAULT_OPTION,
                JOptionPane.PLAIN_MESSAGE,
                null,
                options,
                options[0]);

        // Wykonanie obliczeń na podstawie wybranej opcji
        double result = 0;
        switch (choice) {
            case 0:
                result = number1 + number2;
                JOptionPane.showMessageDialog(null, "Wynik dodawania: " + result);
                break;
            case 1:
                result = number1 - number2;
                JOptionPane.showMessageDialog(null, "Wynik odejmowania: " + result);
                break;
            case 2:
                result = number1 * number2;
                JOptionPane.showMessageDialog(null, "Wynik mnożenia: " + result);
                break;
            case 3:
                if (number2 != 0) {
                    result = number1 / number2;
                    JOptionPane.showMessageDialog(null, "Wynik dzielenia: " + result);
                } else {
                    JOptionPane.showMessageDialog(null, "Błąd: Nie można dzielić przez zero!");
                }
                break;
            default:
                JOptionPane.showMessageDialog(null, "Błąd: Nieprawidłowy wybór!");
                break;
        }
    }
}
